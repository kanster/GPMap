162 template <typename PointT>
163 typename pcl::PointCloud<PointT>::Ptr
164 rangeRemoval(typename pcl::PointCloud<PointT>::ConstPtr cloud,
165  const PointT &origin,
166  const float range,
167  const bool bRemoveClose = true)
168 {
169  // build the condition which a given point must satisfy for it to remain in the point cloud
170  // (p - q)'(p - q) > r^2
171  // => p'p - 2q'p + q'q - r^2 > 0
172  // => p'Ap + 2v'p + c [OP] 0
173  // : A = eye(3), v = -q, c = q'q - r^2, OP = >
174  pcl::ConditionAnd<PointT>::Ptr quad_cond(new pcl::ConditionAnd<PointT> ());
175  quad_cond->addComparison(pcl::TfQuadraticXYZComparison<PointT>::ConstPtr(
176  new pcl::TfQuadraticXYZComparison<PointT>(bRemoveClose ? pcl::ComparisonOps::GT : pcl::ComparisonOps::LT,
177  Eigen::Matrix3f::Identity(),
178  Eigen::Vector3f(-origin.x, -origin.y, -origin.z),
179  (origin.x)*(origin.x) + (origin.y)*(origin.y) + (origin.z)*(origin.z) - range*range)));
180 
181  // build the filter
182  pcl::ConditionalRemoval<PointT> condrem (quad_cond);
183  condrem.setInputCloud (cloud);
184  //condrem.setKeepOrganized(true);
185 
186  // apply filter
187  pcl::PointCloud<PointT>::Ptr cloud_filtered(new pcl::PointCloud<PointT>());
188  condrem.filter (*cloud_filtered);
