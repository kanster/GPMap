316 inline void unitBackRayVector(const PointT &hitPoint,
317  const pcl::PointXYZ &sensorPosition,
318  pcl::PointNormal &P_O)
319 {
320  // check finite
321  assert(pcl::isFinite<PointT>(hitPoint) && pcl::isFinite<pcl::PointXYZ>(sensorPosition));
322 
323  // hit point
324  P_O.x = hitPoint.x;
325  P_O.y = hitPoint.y;
326  P_O.z = hitPoint.z;
327 
328  // vector from a hit point to the origin (sensorPosition)
329  P_O.normal_x = sensorPosition.x - hitPoint.x;
330  P_O.normal_y = sensorPosition.y - hitPoint.y;
331  P_O.normal_z = sensorPosition.z - hitPoint.z;
332  const float length = sqrt(P_O.normal_x*P_O.normal_x + P_O.normal_y*P_O.normal_y + P_O.normal_z*P_O.normal_z);
333 
334  // check length
335  assert(length > std::numeric_limits<float>::epsilon());
336 
337  // normalize
338  P_O.normal_x /= length;
339  P_O.normal_y /= length;
340  P_O.normal_z /= length;
341 
342  // curvature

344 }
