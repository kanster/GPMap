72  void keyboardEventOccurred(const pcl::visualization::KeyboardEvent &event, void *)
73  {
74  if (event.getKeySym() == "c" && event.keyDown()) { m_fDrawWithCubesOrCenterPoints = !m_fDrawWithCubesOrCenterPoints; update(); }
75  else if (event.getKeySym() == "p" && event.keyDown()) { m_fDisplayOriginalPointsWithCubes = !m_fDisplayOriginalPointsWithCubes; update(); }
76  else if (event.getKeySym() == "w" && event.keyDown()) { if(!m_fWireframe) m_fWireframe = true; update(); }
77  else if (event.getKeySym() == "s" && event.keyDown()) { if(m_fWireframe) m_fWireframe = false; update(); }
78  else if (event.getKeySym() == "l" && event.keyDown()) { m_occupancyThreshold += 0.02; extractCenterPoints(); }
79  else if (event.getKeySym() == "k" && event.keyDown()) { m_occupancyThreshold -= 0.02; extractCenterPoints(); }
80  }

220  bool isNotIsolatedVoxel(const pcl::octree::OctreeKey &key)
221  {
222  // if it is on the boundary
223  // TODO: maxKey_
224  if(key.x == 0 || key.y == 0 || key.z == 0) return true;
225 
226  // check if the node is surrounded with occupied nodes
227  if(!m_octree.existLeaf(key.x+1, key.y, key.z )) return true;
228  if(!m_octree.existLeaf(key.x-1, key.y, key.z )) return true;
229  if(!m_octree.existLeaf(key.x, key.y+1, key.z )) return true;
230  if(!m_octree.existLeaf(key.x, key.y-1, key.z )) return true;
231  if(!m_octree.existLeaf(key.x, key.y, key.z+1)) return true;
232  if(!m_octree.existLeaf(key.x, key.y, key.z-1)) return true;
233 
234  return false;
