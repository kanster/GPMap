115  std::vector<std::vector<int> > &partitionedIndices,
116  const bool fSuffling = true)

118  // if the maximum limit is less or equal to zero, do not divide!
119  if(M <= 0) return false;
120 
121  // size
122  const int N = indices.size();
123  if(N <= M) return false;
124 
125  // suffled indices
126  std::vector<int> suffledIndices(indices);
127  if(fSuffling) std::random_shuffle(suffledIndices.begin(), suffledIndices.end());
128 
129  // partitioning
130  partitionedIndices.clear();
131  int from(0), to(-1);
132  while(to < N-1)
133  {
134  // range

136  to = (N-1 < from + M - 1) ? N-1 : from + M - 1;
137 
138  // copy subset
139  partitionedIndices.push_back(std::vector<int>(suffledIndices.begin() + from, suffledIndices.begin() + to + 1));
